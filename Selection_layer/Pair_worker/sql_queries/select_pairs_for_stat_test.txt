SELECT
    id,
    uuid,
    asset_1,
    asset_2,
    liq_5min_mean,
    liq_1h_mean,
    status,
    stat_state,
    bt_state,
    dt_stat_tested
FROM pair_universe
WHERE asset_1 IN ({asset_pool_placeholders})
  AND asset_2 IN ({asset_pool_placeholders})
  AND (
        stat_state = 'new'
        OR (
            stat_state = 'fail'
            AND {allow_stat_retry} = 1
            AND (
                dt_stat_tested IS NULL
                OR dt_stat_tested < NOW() - INTERVAL {stat_retry_cooldown_hours} HOUR
            )
        )
      )
ORDER BY
    dt_stat_tested ASC,
    id ASC
LIMIT {stat_batch_size};
