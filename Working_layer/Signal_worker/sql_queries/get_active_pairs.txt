SELECT
    st.uuid,
    st.asset_1,
    st.asset_2,
    st.tp,
    st.sl,
    st.last_z_score AS prev_last_z_score,
    st.signal_this_month,
    st.signal_prev_month,
    st.last_update_ts AS signal_last_update_ts,
    ps.adf,
    ps.p_value,
    ps.beta_norm,
    ps.level_30,
    ps.level_180,
    ps.quarantine_until
FROM signal_table st
LEFT JOIN pair_state ps
    ON ps.uuid = st.uuid
ORDER BY st.uuid;
